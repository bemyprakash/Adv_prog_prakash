@startuml OrderSequence

actor Customer
participant "Order Service"
participant Restaurant
participant "Delivery Agent"
database Database

Customer -> "Order Service" : place_order()
"Order Service" -> Database : create_order(PENDING)
"Order Service" --> Customer : order_id

Restaurant -> "Order Service" : accept_order()
"Order Service" -> Database : update(CONFIRMED)
"Order Service" -> Database : update(PREPARING)

"Delivery Agent" -> "Order Service" : accept_delivery()
"Order Service" -> Database : assign_agent()
"Order Service" -> Database : OUT_FOR_DELIVERY

"Delivery Agent" -> "Order Service" : mark_delivered()
"Order Service" -> Database : DELIVERED

"Order Service" -> Customer : notify()

@enduml





@startuml OrderState

[*] --> PENDING
PENDING --> CANCELLED
PENDING --> CONFIRMED
CONFIRMED --> PREPARING
PREPARING --> OUT_FOR_DELIVERY
OUT_FOR_DELIVERY --> DELIVERED

CANCELLED --> [*]
DELIVERED --> [*]

@enduml
